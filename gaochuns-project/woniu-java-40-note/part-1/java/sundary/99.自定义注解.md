
```java
@Target(ElementType.TYPE)
@Retention(RetentionPolicy.SOURCE)
public @interface JavaBean {
    JavaBeanType type() default JavaBeanType.Unknown;
}

public enum JavaBeanType {
    /**
     * 默认值
     */
    Unknown,
    /**
     * 持久化层使用的 java bean 。
     * 持久化对象，和数据库中的表一一对应。
     */
    PersistenceObject,

    /**
     * 业务层使用的 java bean。
     * 领域对象。一般中小型项目中用不上它。
     */
    Domain,

    /**
     * web 层使用的 java bean 。
     * 表单对象，和页面的数据提交相关。
     */
    FormObject,

    /**
     * web 层使用的 java bean
     * 数据传输对象，页面返回给请求发起方时用的对象。
     */
    Dto
}
```
